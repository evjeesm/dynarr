313void vector_spread(vector_t *const vector, const size_t index, const size_t amount)
314{
315 assert(vector);
316 assert(amount > 1);
317 assert((index < vector->capacity) && "Index out of capacity bounds!");
318 assert((index + amount <= vector->capacity) && "`index + amount` exceedes vector's capacity.");
319
320 size_t elements_set = 1;
321
322 /* copy elements exponentially through out the range */
323 for (; (elements_set + elements_set) <= amount; elements_set += elements_set)
324 {
325 void *dest = vector_get(vector, index + elements_set);
326 vector_copy(vector, dest, index, elements_set);
327 }
328 /* copy rest that left */
329 for (; elements_set < amount; ++elements_set)
330 {
331 void *dest = vector_get(vector, index + elements_set);
332 vector_copy(vector, dest, index + elements_set - 1, 1);
333 }
334}

461static void *binary_find(const vector_t *const vector,
462 const void *const value,
463 const size_t start,
464 const size_t end,
465 const compare_t cmp,
466 void *param)
467{
468 if (start == end)
469 {
470 return NULL;
471 }
472
473 const size_t middle = (start + end) / 2;
474 void *middle_value = vector_get(vector, middle);
475
476 if (0 == cmp(value, middle_value, param))
477 {
478 return middle_value;
479 }
480
481 if (0 < cmp(value, middle_value, param))
482 {
483 return binary_find(vector, value, middle + 1, end, cmp, param);
484 }
485
486 return binary_find(vector, value, start, middle, cmp, param);
487}
