87dynarr_t *dynarr_create_(const dynarr_opts_t *const opts)
88{
89 assert(opts);
90 assert(opts->element_size);
91
92 const size_t grow_at = ceil(opts->initial_cap * opts->grow_threshold);
93 const size_t next_shrink_at = floor(grow_at * opts->shrink_threshold);
94 assert(next_shrink_at <= grow_at);
95
96 dynarr_t *dynarr = vector_create(
97 .data_offset = sizeof(dynarr_header_t) + opts->data_offset,
98 .element_size = opts->element_size,
99 .initial_cap = opts->initial_cap,
100 .alloc_param = opts->alloc_param,
101 );
102
103 if (NULL == dynarr) return NULL;
104
105 dynarr_header_t *header = get_dynarr_header(dynarr);
106 *header = (dynarr_header_t){
107 .size = 0,
108 opts->grow_factor,
109 opts->grow_threshold,
110 opts->shrink_threshold
111 };
112
113 return dynarr;
114}

413dynarr_t *dynarr_binary_merge(const dynarr_t *const first,
414 const dynarr_t *const second,
415 const compare_t cmp,
416 void *const param)
417{
418 assert(first);
419 assert(second);
420 assert(cmp);
421
422 // copy opts from first dynarr
423 // NOTE: maybe its relevant to allow for preallocation without using .initial_cap,
424 // like dynarr_create_prealoc(opts, prealoc_size) ?
425 dynarr_t *merged = dynarr_create_(TMP_REF(dynarr_opts_t, get_opts(first)));
426 if (!merged) return NULL;
427
428 dynarr_status_t status = 0;
429 const size_t first_size = dynarr_size(first);
430 const size_t second_size = dynarr_size(second);
431 size_t first_i = 0;
432 size_t second_i = 0;
433
434 for (; first_i < first_size && second_i < second_size ;)
435 {
436 const void *f = dynarr_get(first, first_i);
437 const void *s = dynarr_get(second, second_i);
438 ssize_t res = cmp(f, s, param);
439 if (0 == res)
440 {
441 status = dynarr_append(&merged, f);
442 ++first_i; ++second_i;
443 }
444 else if (0 > res)
445 {
446 status = dynarr_append(&merged, f);
447 ++first_i;
448 }
449 else // 0 < res
450 {
451 status = dynarr_append(&merged, s);
452 ++second_i;
453 }
454
455 if (DYNARR_SUCCESS != status)
456 {
457 dynarr_destroy(merged);
458 return NULL;
459 }
460 }
461
462 if (first_size == second_size) return merged;
463
464 // copy rest
465 {
466 const dynarr_t *src;
467 size_t src_i;
468 size_t src_size;
469 if (first_size > second_size)
470 {
471 src = first;
472 src_i = first_i;
473 src_size = first_size;
474 }
475 else {
476 src = second;
477 src_i = second_i;
478 src_size = second_size;
479 }
480
481 for (; src_i < src_size; ++src_i)
482 {
483 status = dynarr_append(&merged, dynarr_get(src, src_i));
484 if (DYNARR_SUCCESS != status)
485 {
486 dynarr_destroy(merged);
487 return NULL;
488 }
489 }
490 }
491
492 return merged;
493}

552static dynarr_status_t grow(dynarr_t **const dynarr, const size_t amount_to_add)
553{
554 const dynarr_header_t *const header = get_dynarr_header(*dynarr);
555 const size_t size = dynarr_size(*dynarr) + amount_to_add;
556 double capacity = dynarr_capacity(*dynarr);
557 size_t grow_at = capacity * header->grow_threshold;
558
559 while (size >= grow_at)
560 {
561 capacity *= header->grow_factor;
562 grow_at = capacity * header->grow_threshold;
563 }
564
565 const size_t new_cap = ceil(capacity);
566 if (new_cap == dynarr_capacity(*dynarr))
567 {
568 return DYNARR_SUCCESS;
569 }
570
571 return (dynarr_status_t)vector_resize(dynarr, new_cap, (vector_status_t)DYNARR_GROW_ERROR);
572}

575static dynarr_status_t shrink(dynarr_t **const dynarr, const size_t amount_to_remove)
576{
577 const dynarr_header_t *const header = get_dynarr_header(*dynarr);
578 const size_t size = dynarr_size(*dynarr) - amount_to_remove;
579 double capacity = dynarr_capacity(*dynarr);
580 size_t shrink_at = capacity * header->shrink_threshold;
581
582 while (size < shrink_at)
583 {
584 capacity /= header->grow_factor;
585 shrink_at = capacity * header->shrink_threshold;
586 }
587
588 const size_t initial_cap = vector_initial_capacity(*dynarr);
589 size_t new_cap = floor(capacity);
590 new_cap = new_cap < initial_cap ? initial_cap : new_cap;
591
592 if (dynarr_capacity(*dynarr) == new_cap)
593 {
594 return DYNARR_SUCCESS;
595 }
596
597 return (dynarr_status_t)vector_resize(dynarr, new_cap, (vector_status_t)DYNARR_SHRINK_ERROR);
598}

601static void free_space_at(dynarr_t *const dynarr, const size_t index, const size_t amount)
602{
603 const size_t size = dynarr_size(dynarr);
604 dynarr_header_t *header = get_dynarr_header(dynarr);
605
606 if (index >= size)
607 {
608 return; // already free
609 }
610
611 const size_t rest_from_index = size - index;
612
613 if (amount >= rest_from_index)
614 {
615 header->size -= rest_from_index;
616 }
617 else
618 {
619 const size_t element_size = vector_element_size(dynarr);
620 size_t rest = rest_from_index - amount;
621 char *to = (char*) dynarr_get(dynarr, 0) + element_size * index;
622 char *from = to + element_size * amount;
623
624 memmove(to, from, element_size * rest);
625 header->size -= amount;
626 }
627}

630static void make_space_at(dynarr_t *const dynarr, const size_t index, size_t amount)
631{
632 const size_t size = dynarr_size(dynarr);
633 dynarr_header_t *header = get_dynarr_header(dynarr);
634
635 if (index > size)
636 {
637 amount += index - size;
638 }
639
640 if (index < size)
641 {
642 const size_t element_size = vector_element_size(dynarr);
643 char *from = (char*) dynarr_get(dynarr, 0) + element_size * index;
644 char *to = from + element_size * amount;
645
646 memmove(to, from, element_size * (size - index));
647 }
648
649 header->size += amount;
650}

653static size_t binary_find_insert_place(const dynarr_t *const dynarr,
654 const void *value,
655 const size_t start,
656 const size_t end,
657 const compare_t cmp,
658 void *param)
659{
660 if (start == end)
661 {
662 return start;
663 }
664
665 const size_t middle = (start + end) / 2;
666 const void *middle_value = dynarr_get(dynarr, middle);
667
668 if (0 == cmp(value, middle_value, param))
669 {
670 return middle + 1;
671 }
672
673 if (0 < cmp(value, middle_value, param))
674 {
675 return binary_find_insert_place(dynarr, value, middle + 1, end, cmp, param);
676 }
677
678 return binary_find_insert_place(dynarr, value, start, middle, cmp, param);
679}
